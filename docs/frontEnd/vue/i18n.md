## 使用 I18n 实现国际化
i18n 全称为 internationalization 国际化。 i和n 表示首位字母，18表示中间的字符数。
```
npm install vue
npm install vue-i18n

// main.js
import Vue from 'vue'
import VueI18n from 'vue-i18n';

const i18n = createI18n({
  locale: 'ja',
  messages: {
    'zh': require('@/assets/lang/zh.js'),
    'en': require('@/assets/lang/en.js')
  }
})

Vue.use(VueI18n);

// en.js
export const lang = {
  login: 'login',
  register: 'register',
  rember: 'rember',
  home: 'home',
  about: 'about',
  aboutDesc: 'This is an about page'
}

// zh.js
export const lang = {
  login: '登录',
  register: '注册',
  rember: '记住密码',
  home: '首页',
  about: '关于',
  aboutDesc: '这是关于页面'
}

// 页面使用
{{$t('lang.login')}}

// 获取当前语言
this.$i18n.locale

// 切换语言 lang
this.$i18n.locale = lang;
```

## vue-i18n-loader
main.js中引入国际化资源的方式会给开发带来不变，需要在开发页面的同时去配置国际化资源。所以如果可以在每一个组件中维护国际化资源，开发时就比较友好了。

### `<i18n>` 组件代码块
```
npm i --save-dev @intlify/vue-i18n-loader

// webpack.conf.js 配置中增加 loader
rules: [
  {
    resourceQuery: /blockType=i18n/,
    type: 'javascript/auto',
    loader: '@intlify/vue-i18n-loader',
  },
]

// 组件中单独维护国际化资源
<i18n>
{
  "en": {
    priviteMsg: 'hello world',
  },
  
  "zh": {
    priviteMsg: '你好世界',
  }
}
</i18n>

// 使用方式
{{ $t(priviteMsg) }}
```

### json 文件
```
rules: [
  {
    test: /\.(json5?|ya?ml)$/, // target json, json5, yaml and yml files
    type: 'javascript/auto',
    loader: '@intlify/vue-i18n-loader',
    include: [ // Use `Rule.include` to specify the files of locale messages to be pre-compiled
      path.resolve(__dirname, 'src/locales')
    ]
  }
]
// main.js
import Vue from 'vue'
import VueI18n from 'vue-i18n';

const i18n = createI18n({
  locale: 'ja',
  messages: {
    'zh': require('@/assets/lang/zh.json'),
    'en': require('@/assets/lang/en.json')
  }
})

Vue.use(VueI18n);

// en.json
{
  login: 'login',
  register: 'register',
  rember: 'rember',
  home: 'home',
  about: 'about',
  aboutDesc: 'This is an about page'
}

// zh.json
{
  login: '登录',
  register: '注册',
  rember: '记住密码',
  home: '首页',
  about: '关于',
  aboutDesc: '这是关于页面'
}

// 页面使用
{{$t('login')}}
```
